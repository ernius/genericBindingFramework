data freshF (S : Sort)(n : V){F : Functor} : (G : Functor) → ⟦ G ⟧ (μ F) → Set where
   freshV≢   : {m : V} → n ≢ m    → freshF S n (|v| S) m
   freshV≢S  : {m : V}{S' : Sort} 
             → S' ≢ S             → freshF S n (|v| S') m
   fresh1    :                      freshF S n |1|     tt
   freshE    : {B : Set}{b : B}   → freshF S n (|E| B) b
   freshEf   : {G : Functor}{e : ⟦ G ⟧ (μ G)}
             → freshF S n G e     → freshF S n (|Ef| G ) ⟨ e ⟩
   freshR    : {e : ⟦ F ⟧ (μ F)}
             → freshF S n F e     → freshF S n |R|       ⟨ e ⟩            
   freshinj₁ : {F₁ F₂ : Functor}{e : ⟦ F₁ ⟧ (μ F)}
             → freshF S n F₁ e    → freshF S n (F₁ |+| F₂) (inj₁ e) 
   freshinj₂ : {F₁ F₂ : Functor}{e : ⟦ F₂ ⟧ (μ F)}
             → freshF S n F₂ e    → freshF S n (F₁ |+| F₂) (inj₂ e) 
   freshx    : {F₁ F₂ : Functor}{e₁ : ⟦ F₁ ⟧ (μ F)}{e₂ : ⟦ F₂ ⟧ (μ F)}
             → freshF S n F₁ e₁
             → freshF S n F₂ e₂   → freshF S n (F₁ |x| F₂) (e₁ , e₂) 
   freshb≡   : {G : Functor}{e : ⟦ G ⟧ (μ F)}
                                  → freshF S n (|B| S G) (n , e)
   freshb    : {G : Functor}{S' : Sort}{e : ⟦ G ⟧ (μ F)}{m : V}
             → freshF S n G e     → freshF S n (|B| S' G) (m , e)

fresh : (S : Sort)(n : V){F : Functor} → (μ F) → Set
fresh S n {F} = freshF S n |R| -- ⟨ e ⟩ = freshF S n {F} F e
