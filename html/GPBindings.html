\begin{code}
data Functor : Set₁ where               
  |v|    : Sort               →  Functor
  |1|    :                       Functor
  |R|    :                       Functor
  |E|    : Set                →  Functor        
  |Ef|   : Functor            →  Functor        
  _|+|_  : Functor → Functor  →  Functor     
  _|x|_  : Functor → Functor  →  Functor     
  |B|    : Sort → Functor     →  Functor
\end{code}

mutual
  ⟦_⟧ : Functor → Set → Set           
  ⟦ |v|   _  ⟧ _  = ℕ
  ⟦ |1|      ⟧ _  = ⊤
  ⟦ |E|   B  ⟧ _  = B
  ⟦ |Ef|  F  ⟧ _  = μ F
  ⟦ |R|      ⟧ A  = A
  ⟦ F |+| G  ⟧ A  = ⟦ F ⟧ A  ⊎ ⟦ G ⟧ A
  ⟦ F |x| G  ⟧ A  = ⟦ F ⟧ A  × ⟦ G ⟧ A 
  ⟦ |B| S G  ⟧ A  = ℕ        × ⟦ G ⟧ A

  data μ (F : Functor) : Set where
    ⟨_⟩ : ⟦ F ⟧ (μ F) → μ F

%<*foldmap>
\begin{code}
foldmap :  {A : Set}(F G  : Functor)
           → (⟦ F ⟧ A → A) → ⟦ G ⟧ (μ F) → ⟦ G ⟧ A
foldmap F (|v| S)      f a         = a
foldmap F |1|          f e         = e
foldmap F (|E| A)      f e         = e
foldmap F (|Ef| G)     f e         = e
foldmap F |R|          f ⟨ e ⟩     = f     (foldmap F F f e) 
foldmap F (G₁ |+| G₂)  f (inj₁ x)  = inj₁  (foldmap F G₁ f x)
foldmap F (G₁ |+| G₂)  f (inj₂ y)  = inj₂  (foldmap F G₂ f y)
foldmap F (G₁ |x| G₂)  f (x , y)   = foldmap F G₁ f x  , foldmap F G₂ f y
foldmap F (|B| S G)    f (x , y )  = x                 , foldmap F G f y

fold : {A : Set}(F : Functor) → (⟦ F ⟧ A → A) → μ F → A
fold F f e = foldmap F |R| f e 
\end{code}

%<*foldmapCtx>
\begin{code}
foldmapCtx :  {C H : Functor}(F G  : Functor)
              → (μ C → ⟦ F ⟧ (μ H)  → μ H)
              → μ C →  ⟦ G ⟧ (μ F) → ⟦ G ⟧ (μ H)
foldmapCtx F (|v| S)      f c a         = a
foldmapCtx F |1|          f c e         = e
foldmapCtx F (|E| A)      f c e         = e
foldmapCtx F (|Ef| G)     f c e         = e
foldmapCtx F |R|          f c ⟨ e ⟩     = f c   (foldmapCtx F F f c e) 
foldmapCtx F (G₁ |+| G₂)  f c (inj₁ x)  = inj₁  (foldmapCtx F G₁ f c x)
foldmapCtx F (G₁ |+| G₂)  f c (inj₂ y)  = inj₂  (foldmapCtx F G₂ f c y)
foldmapCtx F (G₁ |x| G₂)  f c (x , y)   = foldmapCtx F G₁ f c x  , foldmapCtx F G₂ f c y
foldmapCtx F (|B| S G)    f c (x , y)   = x                      , foldmapCtx F G f c y

foldCtx :  {C H : Functor}(F : Functor)
           → (μ C → ⟦ F ⟧ (μ H) → μ H)
           → μ C → μ F → μ H
foldCtx {C} {H} F f c e = foldmapCtx {C} {H} F |R| f c e  
\end{code}
%</foldmapCtx>

Primitive Induction

%<*primInd>
\begin{code}
fih : {F : Functor}(G : Functor)(P : μ F → Set) → ⟦ G ⟧ (μ F) → Set
fih (|v| S)      P n         = ⊤
fih |1|          P tt        = ⊤
fih (|E| B)      P b         = ⊤
fih (|Ef| G)     P e         = ⊤
fih |R|          P e         = P e
fih (G₁ |+| G₂)  P (inj₁ x)  = fih G₁   P x
fih (G₁ |+| G₂)  P (inj₂ y)  = fih G₂  P y 
fih (G₁ |x| G₂)  P (x , y)   = fih G₁  P x × fih G₂ P y
fih (|B| S G)    P (x , y)   = fih G   P y

foldmapFh :  {F : Functor}(G : Functor)(P : μ F → Set)
             → ((e : ⟦ F ⟧ (μ F)) → fih F P e →  P ⟨ e ⟩)
             → (x : ⟦ G ⟧ (μ F)) → fih G P x
foldmapFh (|v| S)      P hi n         = tt
foldmapFh |1|          P hi tt        = tt
foldmapFh (|E| B)      P hi b         = tt
foldmapFh (|Ef| F)     P hi b         = tt
foldmapFh {F} |R|      P hi ⟨ e ⟩     = hi e (foldmapFh {F} F P hi e) 
foldmapFh (G₁ |+| G₂)  P hi (inj₁ x)  = foldmapFh G₁  P hi x
foldmapFh (G₁ |+| G₂)  P hi (inj₂ y)  = foldmapFh G₂  P hi y
foldmapFh (G₁ |x| G₂)  P hi (x , y)   = foldmapFh G₁  P hi x , foldmapFh G₂ P hi y
foldmapFh (|B| S G)    P hi (x , y)   = foldmapFh G   P hi y

foldInd :  (F : Functor)(P : μ F → Set)    
           → ((e : ⟦ F ⟧ (μ F)) → fih F P e →  P ⟨ e ⟩) → (e : μ F) → P e
foldInd F P hi e = foldmapFh {F} |R| P hi e
\end{code}
%</primInd>

Size

\begin{code}
sizeF : {F : Functor}(G : Functor) → ⟦ G ⟧ (μ F) → ℕ
sizeF (|v| x)     e          = 1
sizeF |1|         e          = 1
sizeF (|E| x)     e          = 1
sizeF (|Ef| G)    ⟨ e ⟩      = sizeF G e
sizeF {F} |R|     ⟨ e ⟩      = sizeF F e
sizeF (G₁ |+| G₂) (inj₁ e)   = sizeF G₁ e
sizeF (G₁ |+| G₂) (inj₂ e)   = sizeF G₂ e
sizeF (G₁ |x| G₂) (e₁ , e₂)  = sizeF G₁ e₁ + sizeF G₂ e₂
sizeF (|B| S G)   (x  , y)   = suc (sizeF G y)

size : {F : Functor} → μ F → ℕ
size {F} e = sizeF |R| e

open import Data.Nat.Properties
open DecTotalOrder Data.Nat.decTotalOrder using () renaming (refl to ≤-refl)
open ≤-Reasoning
  renaming (begin_ to start_; _∎ to _◽; _≡⟨_⟩_ to _≤⟨_⟩'_)

lemmasizeF>0 : {F G : Functor}(e : ⟦ G ⟧ (μ F)) → sizeF G e > 0
lemmasizeF>0 {G = |v| S}      e         = s≤s z≤n
lemmasizeF>0 {G = |1|}        e         = s≤s z≤n
lemmasizeF>0 {F} {|R|}        ⟨ e ⟩     =  lemmasizeF>0 {F} {F} e
lemmasizeF>0 {G = |E| x}      e         = s≤s z≤n
lemmasizeF>0 {G = |Ef| G}     ⟨ e ⟩     = lemmasizeF>0 {G} {G} e
lemmasizeF>0 {G = G₁ |+| G₂}  (inj₁ e)  = lemmasizeF>0 {G = G₁} e
lemmasizeF>0 {G = G₁ |+| G₂}  (inj₂ e)  = lemmasizeF>0 {G = G₂} e
lemmasizeF>0 {G = G₁ |x| G₂}  (e₁ , e₂) =   
  start
    suc zero
  ≤⟨ lemmasizeF>0 {G = G₁} e₁ ⟩
    sizeF G₁ e₁
  ≤⟨ m≤m+n (sizeF G₁ e₁) (sizeF G₂ e₂) ⟩
    (sizeF G₁ e₁) + (sizeF G₂ e₂) 
  ≤⟨ ≤-refl ⟩
    sizeF (G₁ |x| G₂) (e₁ , e₂)
  ◽
lemmasizeF>0 {G = |B| S G}    (x , e)    =
  start
    suc zero 
  ≤⟨ s≤s z≤n ⟩
    suc (suc zero)
  ≤⟨ s≤s (lemmasizeF>0 {G = G} e) ⟩
    1 + sizeF G e
  ≤⟨ ≤-refl ⟩
    sizeF (|B| S G) (x , e)
  ◽
\end{code}
