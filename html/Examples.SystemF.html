SortTypeVars : Sort
SortTypeVars = 1

SortTermVars : Sort
SortTermVars = 2

tyF : Functor                               -- type :-
tyF =      |v| SortTypeVars                 --   α
      |+|  |R| |x| |R|                      -- | type ⟶ type
      |+|  |B| SortTypeVars |R|             -- | ∀ α . type

tF : Functor                                -- term :-
tF =      |v| SortTermVars                  --   x
     |+|  |R| |x| |R|                       -- | term term
     |+|  |Ef| tyF |x| |B| SortTermVars |R| -- | λ x : type . term
     |+|  |R| |x| |Ef| tyF                  -- | term type
     |+|  |B| SortTypeVars |R|              -- | Λ α . term

-- Define function with fold
varsaux : ⟦ tF ⟧ V → V
varsaux (inj₁ n)                            = 1
varsaux (inj₂ (inj₁ (m , n)))               = m + n
varsaux (inj₂ (inj₂ (inj₁ (_ , _ , m))))    = m
varsaux (inj₂ (inj₂ (inj₂ (inj₁ (n , _))))) = n
varsaux (inj₂ (inj₂ (inj₂ (inj₂ (_ , m))))) = m

vars : Term → V
vars = fold tF varsaux

-- Prove trivial property of vars function by primitive induction.
PVars : Term → Set
PVars M = vars M > 0

ProvePVars : (M : Term) → PVars M
ProvePVars = foldInd tF PVars proof
   where
   plus>0 : {m n : ℕ} → m > 0 → n > 0 → m + n > 0
   plus>0 {m} {n} m>0 n>0 = ≤-steps m n>0
   proof : (e : ⟦ tF ⟧ (μ tF)) → fih tF PVars e → PVars ⟨ e ⟩
   proof (inj₁ x)                                 tt         = s≤s z≤n
   proof (inj₂ (inj₁ (⟨ M ⟩ , ⟨ N ⟩)))            (PM , PN)  = plus>0 PM PN
   proof (inj₂ (inj₂ (inj₁ (α , x , ⟨ M ⟩))))     (tt , PM)  = PM
   proof (inj₂ (inj₂ (inj₂ (inj₁ (⟨ M ⟩ , α)))))  (PM , tt)  = PM
   proof (inj₂ (inj₂ (inj₂ (inj₂ (x , ⟨ M ⟩)))))  PM         = PM

substaux : μ cF → ⟦ tF ⟧ (μ tF) → μ tF

       (M [ x ≔ N ]n) · (M′ [ x ≔ N ]n)
     ≡⟨ refl ⟩
       (M · M′) [ x ≔ N ]n
     □
lemma-substnfv  {⟨ inj₂ (inj₂ (inj₁ (t , y , M))) ⟩}  {N} {x} x∉fv 
                (notOcurrBR (notOcurrBinj₂ (notOcurrBinj₂  (notOcurrBinj₁ (notOcurrBx _ (notOcurrBB≢ x≢y xnbM)))))) 
  =  begin≡
       ƛ y t M
     ≡⟨ cong (ƛ y t) (lemma-substnfv  {M}
                                      (lemma-∉  (λ y≡x → ⊥-elim (x≢y (sym y≡x)))
                                                (c∉xs++ys→c∉ys {x} {fvSF {tF} {|Ef| tyF} SortTermVars t} x∉fv))
                                      xnbM)  ⟩  
       ƛ y t (M [ x ≔ N ]n)
     ≡⟨ refl ⟩      
        (ƛ y t M) [ x ≔ N ]n
     □
lemma-substnfv  {⟨ inj₂ (inj₂ (inj₂ (inj₁ (M , t)))) ⟩} {N} {x} x∉fv
                (notOcurrBR (notOcurrBinj₂ (notOcurrBinj₂  (notOcurrBinj₂ (notOcurrBinj₁ (notOcurrBx xnbM _))))))
  =  begin≡
        M ·ₜ t
      ≡⟨  cong (λ P →  P ·ₜ t) (lemma-substnfv {M} (c∉xs++ys→c∉xs x∉fv) xnbM) ⟩
        (M [ x ≔ N ]n) ·ₜ t
      ≡⟨ refl ⟩
        (M ·ₜ t) [ x ≔ N ]n
      □
lemma-substnfv  {⟨ inj₂ (inj₂ (inj₂ (inj₂ (α , M)))) ⟩} {N} {x} x∉fv
                (notOcurrBR (notOcurrBinj₂ (notOcurrBinj₂  (notOcurrBinj₂ (notOcurrBinj₂ (notOcurrBB≢ x≢α xnbM))))))
  =  begin≡
       Λ α M
     ≡⟨ cong (Λ α) (lemma-substnfv {M} x∉fv xnbM) ⟩
       Λ α (M [ x ≔ N ]n)
     ≡⟨ refl ⟩
       (Λ α M) [ x ≔ N ]n
     □

PSCn : {x y : V}{L : Term} → Term → Term → Set
PSCn {x} {y} {L} N M = x ∉ y ∷ fv L → x notOcurrBind L → (M [ x ≔ N ]n) [ y ≔ L ]n ~α (M [ y ≔ L ]n)[ x ≔ N [ y ≔ L ]n ]n

lemma-substCompositionV  : {x y z : V}{N L : Term}
                         → x ∉ y ∷ fv  L
                         → x notOcurrBind L
                         → (v z) [ x ≔ N ]n [ y ≔ L ]n ~α (v z) [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n 
lemma-substCompositionV {x} {y} {z} {N} {L} x∉yfvL _ with x ≟v z
... | yes x≡z with y ≟v z 
...         | yes y≡z             = ⊥-elim ((b∉a∷xs→b≢a x∉yfvL) (trans x≡z (sym y≡z)))
...         | no  y≢z with x ≟v z
...                   |  yes _    = ρ
...                   |  no  x≢z  = ⊥-elim (x≢z x≡z)
lemma-substCompositionV {x} {y} {z} {N} {L} x∉yfvL xnbL
    | no  x≢z with y ≟v z 
...         | yes y≡z
  = subst  (λ M →  L ~α M)
           (lemma-substnfv (lemma∉fvfvS {tF} {SortTermVars} {L} (b∉a∷xs→b∉xs x∉yfvL)) xnbL) ρ 
...         | no  y≢z with x ≟v z
...                   |  yes x≡z  = ⊥-elim (x≢z x≡z)
...                   |  no  _    = ρ

-- Esto puede demostrarse genericamente usando fold fusion !
lemma-substCompositionN :  {x y : V}{M N L : Term} → PSCn {x} {y} {L} N M
lemma-substCompositionN {x} {y} {M} {N} {L}
  = foldInd tF (PSCn {x} {y} {L} N)
               lemma-substCompositionNAux
               M
  where
  lemma-substCompositionNAux : (e : ⟦ tF ⟧ (μ tF)) → fih tF (PSCn N) e → PSCn N ⟨ e ⟩  
  lemma-substCompositionNAux (inj₁ z) _ x∉yfvL xnbL = lemma-substCompositionV {x} {y} {z} {N} {L} x∉yfvL xnbL
  lemma-substCompositionNAux (inj₂ (inj₁ (M′  , M″ ))) (hiM′ , hiM″) x∉yfvL xnbL
    =   begin
          (M′ ·  M″ ) [ x ≔ N ]n [ y ≔ L ]n
        ≈⟨ refl ⟩
          ((M′ [ x ≔ N ]n) ·  (M″ [ x ≔ N ]n))  [ y ≔ L ]n
        ≈⟨ refl ⟩
          (M′ [ x ≔ N ]n [ y ≔ L ]n) ·  (M″ [ x ≔ N ]n [ y ≔ L ]n)
        ∼⟨  ~αR (~α+₂ (~α+₁ (~αx (hiM′ x∉yfvL xnbL) (hiM″ x∉yfvL xnbL)))) ⟩
          (M′ [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n ) ·  (M″ [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n)
        ≈⟨ refl ⟩
          (M′ ·  M″) [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n 
        ∎
  lemma-substCompositionNAux (inj₂ (inj₂ (inj₁ (t , z , M)))) (_ , hiM) x∉yfvL xnbL
    =  begin
         (ƛ z t M) [ x ≔ N ]n [ y ≔ L ]n
       ≈⟨ refl ⟩
         ƛ z t (M [ x ≔ N ]n [ y ≔ L ]n)
       ∼⟨ ~αR (~α+₂ (~α+₂ (~α+₁ (~αx ρF (~αB [] (λ u u∉[] → lemma~swapEquiv (hiM x∉yfvL xnbL) z u)))))) ⟩
         ƛ z t (M [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n)
       ≈⟨ refl ⟩
         (ƛ z t M) [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n
       ∎
  lemma-substCompositionNAux (inj₂ (inj₂ (inj₂ (inj₁ (M , t))))) (hi , _) x∉yfvL xnbL
    =  begin
         (M ·ₜ t) [ x ≔ N ]n [ y ≔ L ]n
       ≈⟨ refl ⟩
         (M [ x ≔ N ]n [ y ≔ L ]n) ·ₜ t 
       ∼⟨ ~αR (~α+₂ (~α+₂ (~α+₂ (~α+₁ (~αx (hi x∉yfvL xnbL) ρF))))) ⟩
         (M [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n) ·ₜ t 
       ≈⟨ refl ⟩
         (M ·ₜ t) [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n
       ∎
  lemma-substCompositionNAux (inj₂ (inj₂ (inj₂ (inj₂ (t , M))))) hi x∉yfvL xnbL
    =  begin
          (Λ t M) [ x ≔ N ]n [ y ≔ L ]n
       ≈⟨ refl ⟩
          Λ t (M [ x ≔ N ]n [ y ≔ L ]n)
       ∼⟨ ~αR (~α+₂ (~α+₂ (~α+₂ (~α+₂ (~αB [] (λ c c∉[] → lemma~swapEquiv (hi x∉yfvL xnbL) t c)))))) ⟩
          Λ t (M [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n)
       ≈⟨ refl ⟩
          (Λ t M) [ y ≔ L ]n [ x ≔ N [ y ≔ L ]n ]n
       ∎

PSC : {x y : V}{L : Term} → Term → Term → Set
PSC {x} {y} {L} N M = x ∉ y ∷ fv L
  → (M [ x ≔ N ]) [ y ≔ L ] ~α (M [ y ≔ L ])[ x ≔ N [ y ≔ L ] ]

αCompatiblePSC : ∀ {x y L} N → αCompatiblePred (PSC {x} {y} {L} N)
αCompatiblePSC {x} {y} {L} N {M} {P} M∼P PM x∉y∷fvL --x≢y x∉fvL 
  =  begin
       (P [ x ≔ N ]) [ y ≔ L ] 
     -- Strong α compability of inner substitution operation
     ≈⟨ cong (λ z → z [ y ≔ L ]) (lemma-substα (σ M∼P)) ⟩ 
       (M [ x ≔ N ]) [ y ≔ L ] 
     -- We apply that we know the predicate holds for M
     ∼⟨ PM x∉y∷fvL ⟩ -- x≢y x∉fvL
       (M [ y ≔ L ]) [ x ≔ N [ y ≔ L ] ] 
     -- Strong α compability of inner substitution operation
     ≈⟨ cong (λ z → z [ x ≔ N [ y ≔ L ] ]) (lemma-substα (M∼P))  ⟩ 
       (P [ y ≔ L ]) [ x ≔ N [ y ≔ L ] ] 
     ∎

{- Composition Lemma (exemplifies alpha induction principle usage) -}
lemma-substComposition :  {x y : V}{M N L : Term} → PSC {x} {y} {L} N M
lemma-substComposition {x} {y} {M} {N} {L} 
  = αProof  (PSC {x} {y} {L} N)
            ((y ∷ fv  L′) ++ (x ∷ fv  N) ++ (y ∷ fv  L) ++ (x ∷ fv  (N [ y ≔ L ])))
            SortTermVars
            (αCompatiblePSC  {x} {y} {L} N)
            αproof
            M 
  where
  N′    = proj₁ (bindersFreeαElem (y ∷ fv L) N)
  y:fvLnotBindN′  = proj₂ (bindersFreeαElem (y ∷ fv  L) N)
  N′~N  = lemma-bindersFreeαAlpha (y ∷ fv  L) N
  L′    = proj₁ (bindersFreeαElem (x ∷ fv  (N [ y ≔ L ])) L)
  x:fvN[y:=L]notBindL′  = proj₂ (bindersFreeαElem (x ∷ fv (N [ y ≔ L ])) L)
  L′~L  = lemma-bindersFreeαAlpha (x ∷ fv (N [ y ≔ L ])) L
  y:fvL′notBindN′ = subst (λ xs → ListNotOcurrBind (y ∷ xs) N′) (lemma~fvF  (σ L′~L)) y:fvLnotBindN′
  αproof : (M : Term)
    → ListNotOcurrBind ((y ∷ fv L′) ++ (x ∷ fv  N) ++ (y ∷ fv L)  ++ (x ∷ fv (N [ y ≔ L ]))) M
    → ListNotOcurrBind (fv M) M
    → PSC N M
  αproof M notOcurr _ x∉y∶fvL
    =  begin
         M [ x ≔ N ]    [ y ≔ L ]
       ≈⟨ lemma-substα {M [ x ≔ N ]} (lemmaSubsts {x} {M} {N} x∷fvNnotBindM) ⟩
         M [ x ≔ N ]n   [ y ≔ L ]
       ≈⟨ lemma-substα {M [ x ≔ N ]n} (lemma-substnα′ {x} {M} (σ N′~N)) ⟩
         M [ x ≔ N′ ]n  [ y ≔ L ]
       ∼⟨ lemmaSubsts {y} {M [ x ≔ N′ ]n} {L} nb1 ⟩ 
         M [ x ≔ N′ ]n   [ y ≔ L ]n
       ∼⟨  lemma-substnα′ {y} {M [ x ≔ N′ ]n} (σ L′~L) ⟩
         M [ x ≔ N′ ]n   [ y ≔ L′ ]n
       ∼⟨ lemma-substCompositionN {x} {y} {M} {N′} {L′} x∉y∶fvL′ xNotOcurrBindL′ ⟩
         M [ y ≔ L′ ]n   [ x ≔ N′ [ y ≔ L′ ]n ]n
       ∼⟨  lemma-substnα′ {x} {M [ y ≔ L′ ]n} (σ (lemmaSubsts {y} {N′} y:fvL′notBindN′))  ⟩
         M [ y ≔ L′ ]n   [ x ≔ N′ [ y ≔ L′ ] ]n
       ≈⟨ cong (λ e → M [ y ≔ L′ ]n   [ x ≔ e ]n) (lemma-substα N′~N) ⟩
         M [ y ≔ L′ ]n   [ x ≔ N [ y ≔ L′ ] ]n
       ∼⟨  lemma-substnα′ {x} {M [ y ≔ L′ ]n} (lemma-substα′ {y} {N} L′~L) ⟩
         M [ y ≔ L′ ]n   [ x ≔ N [ y ≔ L  ] ]n
       ∼⟨ σ (lemmaSubsts {x} {M [ y ≔ L′ ]n} {N [ y ≔ L ]} nb2) ⟩
         M [ y ≔ L′ ]n   [ x ≔ N [ y ≔ L ] ]
       ≈⟨ lemma-substα (σ (lemmaSubsts {y} {M} {L′} y∷fvL′notBindM)) ⟩
         M [ y ≔ L′ ]    [ x ≔ N [ y ≔ L ] ]
       ≈⟨ lemma-substα {M [ y ≔ L′ ]} {M [ y ≔ L ]} {N [ y ≔ L ]} {x} (lemma-substα′ {y} {M} {L′} {L} L′~L) ⟩
         M [ y ≔ L ]    [ x ≔ N [ y ≔ L ] ]
       ∎
       where
       xNotOcurrBindL′ : x notOcurrBind L′
       xNotOcurrBindL′ = head x:fvN[y:=L]notBindL′ 
       x∉y∶fvL′ : x ∉ y ∷ fv L′
       x∉y∶fvL′ = subst (λ xs → x ∉ y ∷ xs) (lemma~fvF  (σ L′~L)) x∉y∶fvL 
       x∷fvN[y:=L]notBindM : ListNotOcurrBind (x ∷ fv  (N [ y ≔ L ])) M
       x∷fvN[y:=L]notBindM = lemma-binds++r {y ∷ fv  L} (lemma-binds++r {x ∷ fv  N} (lemma-binds++r {y ∷ fv  L′}  notOcurr))
       y∷fvLnotBindM : ListNotOcurrBind (y ∷ fv  L) M
       y∷fvLnotBindM = lemma-binds++l (lemma-binds++r {x ∷ fv  N} (lemma-binds++r {y ∷ fv  L′}  notOcurr))
       x∷fvNnotBindM : ListNotOcurrBind (x ∷ fv  N) M
       x∷fvNnotBindM = lemma-binds++l (lemma-binds++r {y ∷ fv  L′}  notOcurr)
       y∷fvL′notBindM : ListNotOcurrBind (y ∷ fv  L′) M
       y∷fvL′notBindM = lemma-binds++l notOcurr
       nb1 : ListNotOcurrBind (y ∷ fv  L) (M [ x ≔ N′ ]n)
       nb1 = lemma-foldCtxBinds y∷fvLnotBindM (lemma-bindsR (lemma-binds× lemma-bindsv (lemmalistNotOcurrBindFR→ListNotOcurrBindEf y:fvLnotBindN′))) 
       nb2 : ListNotOcurrBind (x ∷ fv  (N [ y ≔ L ])) (M [ y ≔ L′ ]n)
       nb2 = lemma-foldCtxBinds x∷fvN[y:=L]notBindM (lemma-bindsR (lemma-binds× lemma-bindsv (lemmalistNotOcurrBindFR→ListNotOcurrBindEf x:fvN[y:=L]notBindL′)))
