data notOcurrBindF (n : V){F : Functor} : (G : Functor) → ⟦ G ⟧ (μ F) → Set where
   notOcurrBv     : {m : V}{S : Sort}                          →  notOcurrBindF n (|v| S)     m
   notOcurrB1     :                                               notOcurrBindF n |1|         tt
   notOcurrBE     : {B : Set}{b : B}                           →  notOcurrBindF n (|E| B)     b
   notOcurrBEf    : {G : Functor}{e : ⟦ G ⟧ (μ G)}
                  → notOcurrBindF n G e                        →  notOcurrBindF n (|Ef| G )   ⟨ e ⟩
   notOcurrBR     : {e : ⟦ F ⟧ (μ F)}
                  → notOcurrBindF n F e                        →  notOcurrBindF n |R|         ⟨ e ⟩            
   notOcurrBinj₁  : {F₁ F₂ : Functor}{e : ⟦ F₁ ⟧ (μ F)}
                  → notOcurrBindF n F₁ e                       →  notOcurrBindF n (F₁ |+| F₂) (inj₁ e) 
   notOcurrBinj₂  : {F₁ F₂ : Functor}{e : ⟦ F₂ ⟧ (μ F)}
                  → notOcurrBindF n F₂ e                       →  notOcurrBindF n (F₁ |+| F₂) (inj₂ e) 
   notOcurrBx     : {F₁ F₂ : Functor}{e₁ : ⟦ F₁ ⟧ (μ F)}{e₂ : ⟦ F₂ ⟧ (μ F)}
                  → notOcurrBindF n F₁ e₁
                  → notOcurrBindF n F₂ e₂                      →  notOcurrBindF n (F₁ |x| F₂) (e₁ , e₂) 
   notOcurrBB≢   : {G : Functor}{e : ⟦ G ⟧ (μ F)}{m : V}{S : Sort}
                  → n ≢ m → notOcurrBindF n G e                →  notOcurrBindF n (|B| S G)   (m , e)



_notOcurrBind_ : {F : Functor}(n : V) → (μ F) → Set
_notOcurrBind_ {F} n = notOcurrBindF n {F} |R|

-- Inversion lemmas
notOcurrBinj₁inv : {x : V}{F G₁ G₂ : Functor}{e : ⟦ G₁ ⟧ (μ F)} → notOcurrBindF x (G₁ |+| G₂) (inj₁ e) → notOcurrBindF x G₁ e 
notOcurrBinj₁inv (notOcurrBinj₁ notOcurr) = notOcurr

notOcurrBinj₂inv : {x : V}{F G₁ G₂ : Functor}{e : ⟦ G₂ ⟧ (μ F)} → notOcurrBindF x (G₁ |+| G₂) (inj₂ e) → notOcurrBindF x G₂ e 
notOcurrBinj₂inv (notOcurrBinj₂ notOcurr) = notOcurr

notOcurrBx₁inv : {x : V}{F G₁ G₂ : Functor}{e₁ : ⟦ G₁ ⟧ (μ F)}{e₂ : ⟦ G₂ ⟧ (μ F)} → notOcurrBindF x (G₁ |x| G₂) (e₁ , e₂) → notOcurrBindF x G₁ e₁ 
notOcurrBx₁inv (notOcurrBx notOcurr₁ notOcurr₂) = notOcurr₁

notOcurrBx₂inv : {x : V}{F G₁ G₂ : Functor}{e₁ : ⟦ G₁ ⟧ (μ F)}{e₂ : ⟦ G₂ ⟧ (μ F)} → notOcurrBindF x (G₁ |x| G₂) (e₁ , e₂) → notOcurrBindF x G₂ e₂ 
notOcurrBx₂inv (notOcurrBx notOcurr₁ notOcurr₂) = notOcurr₂

notOcurrBBinv : {x y : V}{F G : Functor}{S : Sort}{e : ⟦ G ⟧ (μ F)} → notOcurrBindF x (|B| S G) (y , e) → notOcurrBindF x G e 
notOcurrBBinv (notOcurrBB≢ _ notOcurr) = notOcurr

notOcurrBRinv : {x : V}{F : Functor}{e : ⟦ F ⟧ (μ F)} → notOcurrBindF x {F} |R| ⟨ e ⟩ → notOcurrBindF x F e 
notOcurrBRinv (notOcurrBR notOcurr) = notOcurr

notOcurrBBinv≢ : {x y : V}{F G : Functor}{S : Sort}{e : ⟦ G ⟧ (μ F)} → notOcurrBindF x (|B| S G) (y , e) → x ≢ y
notOcurrBBinv≢ (notOcurrBB≢ x≢y _) = x≢y

notOcurrBindR→notOcurrBindEf : {F F' : Functor}{e : μ F} → ∀ {x} → notOcurrBindF x {F} |R| e → notOcurrBindF x {F'} (|Ef| F) e
notOcurrBindR→notOcurrBindEf {e = ⟨ e ⟩} notOcurr = notOcurrBEf (notOcurrBRinv notOcurr)

ListNotOcurrBindF : {F : Functor}(G : Functor) → List V → ⟦ G ⟧ (μ F) → Set
ListNotOcurrBindF G xs e = All (λ x → notOcurrBindF x G e) xs

ListNotOcurrBind : {F : Functor} → List V → μ F → Set
ListNotOcurrBind {F} xs e = ListNotOcurrBindF |R| xs e

-- Inversion lemmas
lemmalistNotOcurrBindFR→ListNotOcurrBindEf : {F F' : Functor}{e : μ F}{xs : List V}
  → ListNotOcurrBindF {F} |R| xs e → ListNotOcurrBindF {F'} (|Ef| F) xs e  
lemmalistNotOcurrBindFR→ListNotOcurrBindEf = Data.List.All.map notOcurrBindR→notOcurrBindEf

lemmalistNotOcurrBindFR→ListNotOcurrBindF : {F : Functor}{e : ⟦ F ⟧ (μ F)}{xs : List V}
  → ListNotOcurrBindF {F} |R| xs ⟨ e ⟩ → ListNotOcurrBindF {F} F xs e  
lemmalistNotOcurrBindFR→ListNotOcurrBindF  = Data.List.All.map notOcurrBRinv 

listNotOcurrBinj₁inv : {xs : List V}{F G₁ G₂ : Functor}{e : ⟦ G₁ ⟧ (μ F)}
  → ListNotOcurrBindF (G₁ |+| G₂) xs (inj₁ e) → ListNotOcurrBindF G₁ xs e 
listNotOcurrBinj₁inv  =  Data.List.All.map (λ { (notOcurrBinj₁ nb) → nb})

listNotOcurrBinj₂inv : {xs : List V}{F G₁ G₂ : Functor}{e : ⟦ G₂ ⟧ (μ F)}
  → ListNotOcurrBindF (G₁ |+| G₂) xs (inj₂ e) → ListNotOcurrBindF G₂ xs e 
listNotOcurrBinj₂inv  =  Data.List.All.map (λ { (notOcurrBinj₂ nb) → nb})

listNotOcurrBx₁inv : {xs : List V}{F G₁ G₂ : Functor}{e : ⟦ G₁ ⟧ (μ F)}{e' : ⟦ G₂ ⟧ (μ F)}
  → ListNotOcurrBindF (G₁ |x| G₂) xs (e , e') → ListNotOcurrBindF G₁ xs e 
listNotOcurrBx₁inv  =  Data.List.All.map (λ { (notOcurrBx nb _) → nb})

listNotOcurrBx₂inv : {xs : List V}{F G₁ G₂ : Functor}{e : ⟦ G₁ ⟧ (μ F)}{e' : ⟦ G₂ ⟧ (μ F)}
  → ListNotOcurrBindF (G₁ |x| G₂) xs (e , e') → ListNotOcurrBindF G₂ xs e' 
listNotOcurrBx₂inv  =  Data.List.All.map (λ { (notOcurrBx _ nb) → nb})

get : {F : Functor}{xs : List V}{e : μ F} → ListNotOcurrBind xs e → (x : V) → x ∈ xs → x notOcurrBind e
get nb x x∉xs = lookup nb x∉xs

listNotOcurrBBinv : {xs : List V}{F G : Functor}{x : V}{e : ⟦ G ⟧ (μ F)}{S : Sort}
  → ListNotOcurrBindF (|B| S G) xs (x , e) → ListNotOcurrBindF G xs e
listNotOcurrBBinv = Data.List.All.map (λ { (notOcurrBB≢ _ nb) → nb}) 

listNotOcurrBBinv∉fv : {xs : List V}{F G : Functor}{x : V}{e : ⟦ G ⟧ (μ F)}{S : Sort}
  → ListNotOcurrBindF (|B| S G) xs (x , e) → x ∉ xs
listNotOcurrBBinv∉fv nb x∈xs with lookup nb x∈xs
... | notOcurrBB≢ n≢n _ = ⊥-elim (n≢n refl)

lemma-binds+2 : {F G₁ G₂ : Functor}{e : ⟦ G₂ ⟧ (μ F)}{xs : List V}
                → ListNotOcurrBindF G₂ xs e  → ListNotOcurrBindF (G₁ |+| G₂) xs (inj₂ e)
lemma-binds+2 notOcurr = tabulate (notOcurrBinj₂ ∘ (lookup notOcurr))

lemma-binds+1 : {F G₁ G₂ : Functor}{e : ⟦ G₁ ⟧ (μ F)}{xs : List V}
                → ListNotOcurrBindF G₁ xs e  → ListNotOcurrBindF (G₁ |+| G₂) xs (inj₁ e)
lemma-binds+1 notOcurr = tabulate (notOcurrBinj₁ ∘ (lookup notOcurr))

lemma-binds× : {F G₁ G₂ : Functor}{e : ⟦ G₁ ⟧ (μ F)}{e′ : ⟦ G₂ ⟧ (μ F)}{xs : List V}
                 → ListNotOcurrBindF G₁ xs e → ListNotOcurrBindF G₂ xs e′ → ListNotOcurrBindF (G₁ |x| G₂) xs (e , e′)
lemma-binds× noe noe' = tabulate (λ x∈xs → notOcurrBx ((lookup noe) x∈xs) ((lookup noe') x∈xs))

lemma-binds1 : {F : Functor}{xs : List V}
               → ListNotOcurrBindF {F} |1| xs tt
lemma-binds1 = tabulate (λ _ → notOcurrB1)                 

lemma-bindsE : {F : Functor}{A : Set }{a : A}{xs : List V}
                → ListNotOcurrBindF {F} (|E| A) xs a
lemma-bindsE = tabulate (λ _ → notOcurrBE)

lemma-bindsEf : {F G : Functor}{e : ⟦ F ⟧ (μ F)}{xs : List V}
                 → ListNotOcurrBindF F xs e → ListNotOcurrBindF {G} (|Ef| F) xs ⟨ e ⟩
lemma-bindsEf notOcurr = tabulate (notOcurrBEf ∘ (lookup notOcurr))

lemma-bindsR : {F  : Functor}{e : ⟦ F ⟧ (μ F)}{xs : List V}
                 → ListNotOcurrBindF {F} F xs e → ListNotOcurrBindF {F} |R| xs ⟨ e ⟩
lemma-bindsR notOcurr = tabulate (notOcurrBR ∘ (lookup notOcurr))

lemma-bindsB : {F G  : Functor}{e : ⟦ G ⟧ (μ F)}{xs : List V}{S : Sort}{x : V}
               → x ∉ xs → ListNotOcurrBindF {F} G xs e → ListNotOcurrBindF {F} (|B| S G) xs (x , e)
lemma-bindsB x∉xs notOcurr = tabulate (λ y∈xs → notOcurrBB≢ (x∉xs∧y∈xs⇒y≢x x∉xs y∈xs) ((lookup notOcurr) y∈xs))

lemma-bindsv : {F : Functor}{x : V}{xs : List V}{S : Sort} → ListNotOcurrBindF {F} (|v| S) xs x 
lemma-bindsv = tabulate (λ _ → notOcurrBv)

postulate
  lemma-binds:head  : {x : V}{xs : List V}{F : Functor}{e : μ F} → ListNotOcurrBind (x ∷ xs) e → x notOcurrBind e
  lemma-binds:tail : {x : V}{xs : List V}{F : Functor}{e : μ F} → ListNotOcurrBind (x ∷ xs) e → ListNotOcurrBind xs e
  lemma-binds:cons : {x : V}{xs : List V}{F : Functor}{e : μ F} → x notOcurrBind e → ListNotOcurrBind xs e → ListNotOcurrBind (x ∷ xs) e 
  lemma-binds++l : {xs ys : List V}{F : Functor}{e : μ F} → ListNotOcurrBind (xs ++ ys) e → ListNotOcurrBind xs e
  lemma-binds++r : {xs ys : List V}{F : Functor}{e : μ F} → ListNotOcurrBind (xs ++ ys) e → ListNotOcurrBind ys e
  lemma-binds⊆   : {xs ys : List V}{F : Functor}{e : μ F} → ys ⊆ xs → ListNotOcurrBind xs e → ListNotOcurrBind ys e
  lemma-foldCtxBinds : {C H : Functor}{F : Functor}{f : μ C → ⟦ F ⟧ (μ H) → μ H}{c : μ C}{e : μ F}
                     {xs : List V} → ListNotOcurrBind xs e → ListNotOcurrBind xs c →  ListNotOcurrBind xs (foldCtx F f c e)
