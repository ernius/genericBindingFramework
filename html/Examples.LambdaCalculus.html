\begin{code}
data _⇉n_ : Rel where
  ⇉nv  :  {x : V}
         → v x ⇉n v x
  ⇉nƛ  :  {M N : λTerm}{x : V}
         → M ⇉n N
         → ƛ x M ⇉n ƛ x N
  ⇉n·  :  {M M' N N' : λTerm}
         → M ⇉n M' → N ⇉n N'
         → M · N ⇉n M' · N'
  ⇉nβ :   {M M' N N' : λTerm}{x : V}
         → M ⇉n M' 
         → N ⇉n N'
         → ƛ x M · N ⇉n M' [ x ≔ N' ]ₙ
\end{code}
